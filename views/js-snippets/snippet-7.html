//destructuring parameters
function fullName(user){

}
function fullName(user) {
const {firstName, lastName} = user;
return `${firstName} ${lastName}`
}

function fullName({firstName, lastName}) {
return `${firstName} ${lastName}`
}

const movies = [

},

score: 85,
year: 1983
},
{
title: "parasite",

}
]
